def extract(terms):
    if terms is None:
        ids = set()
    elif type(terms) is dict:
        ids = set([terms["qualifier"]+":"+terms["id"]])
    else:
        ids = set((t["qualifier"]+":"+t["id"] for t in terms))
    return ids
